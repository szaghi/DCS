672  subroutine dcs_simulate()
673  !---------------------------------------------------------------------------------------------------------------------------------
674  implicit none
675  type(type_conservative):: residual
676  integer(I4P):: n = 0_i4p
677  integer(I4P):: i,j
678  !---------------------------------------------------------------------------------------------------------------------------------
679 
680  !---------------------------------------------------------------------------------------------------------------------------------
681  ! successive over-relaxation (SOR) loop
682  call residual%set(s=maxr8p,v=maxr8p)
683  sor: do while(((residual%s(2)>rtol).AND.(residual%v(2)>rtol)))
684  ! updating time step counter
685  n = n + 1_i4p
686  ! updating old conservative variables: cons(n) = cons(n+1)
687  do j=1,cavity%mesh%N-1_i4p
688  do i=1,cavity%mesh%N-1_i4p
689  call cavity%cons(i,j)%oldupdate
690  enddo
691  enddo
692  ! computing new stream function
693  call computestream(cons=cavity%cons,beta=cavity%beta,dh=cavity%mesh%dh)
694  ! computing new vorticity
695  call computevorticity(cons=cavity%cons,re=cavity%Re,beta=cavity%beta,dh=cavity%mesh%dh)
696  ! computing residuals
697  residual = computeresidual(cons=cavity%cons,re=cavity%Re,dh=cavity%mesh%dh)
698  if (mod(n,nout)==0) then
